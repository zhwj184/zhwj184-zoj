int cmp(const void *p1,const void *p2) 
{ 
   Point *p11=(Point *)p1, 
        *p22=(Point *)p2; 
   if(p11->x > p22->x) 
      return 1; 
   if(p11->x == p22->x) 
      return p11->y-p22->y; 
   return -1; 
}
