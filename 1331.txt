#include <stdio.h>

int main(int argc, char *argv[])
{
    int a,b,c,d;
    for(a=6;a<201;a++){
        for(b=2;b<a;b++){
            for(c=b;c<a;c++){
                for(d=c;d<a;d++){
                    if(a*a*a<(b*b*b+c*c*c+d*d*d))
                        break;
                    if(a*a*a==b*b*b+c*c*c+d*d*d)
                        printf("Cube = %d, Triple = (%d,%d,%d)\n",a,b,c,d);
                }
            }
        }
    }
    return 0;
}
